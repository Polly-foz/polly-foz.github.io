Js Array Sort 诡异

Written by Polly-foz with ♥ on February 16, 2020 updateAt Feb 16, 2020 in 
console.log([5,25,4,39,57,49,93,79,7,8,49,89,2,7,73,88,45,15,34,92,84,38,85,34,16,6,99,0,2,36,68,52,73,50,77,44,61,48].sort())

上面代码输出： [ 0, 15, 16, 2, 2, 25, 34, 34, 36, 38, 39, 4, 44, 45, 48, 49, 49, 5, 50, 52, 57, 6, 61, 68, 7, 7, 73, 73, 77, 79, 8, 84, 85, 88, 89, 92, 93, 99 ]

显然 结果乱序了。。。

解决方案

console.log([5,25,4,39,57,49,93,79,7,8,49,89,2,7,73,88,45,15,34,92,84,38,85,34,16,6,99,0,2,36,68,52,73,50,77,44,61,48]
    .sort(function(n,m){
        return n - m;
    }))

[ 0, 2, 2, 4, 5, 6, 7, 7, 8, 15, 16, 25, 34, 34, 36, 38, 39, 44, 45, 48, 49, 49, 50, 52, 57, 61, 68, 73, 73, 77, 79, 84, 85, 88, 89, 92, 93, 99 ]
