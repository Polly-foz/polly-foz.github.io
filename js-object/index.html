JavaScript 对象基本用法

Written by Polly-foz with ♥ on January 1, 2020 in JavaScript

最新的 ECMAScript 标准定义了 8 种数据类型:

  • 7 种原始类型:
    • Boolean
    • Null
    • Undefined
    • Number
    • BigInt
    • String
    • Symbol
  • 和 Object

除 Object 以外的所有类型都是不可变的（值本身无法被改变）。

创建Object实例的两种语法

let obj = {'name':'Peppa','age':3}
let obj = new Object({'name':'Peppa','age':3})
  • 对象是无序的数据集合
  • 对象是键值对的集合，键名是属性名，键值是属性值
  • 键名是字符串，不是标识符，可以包含任意字符
  • 引号可省略，省略之后就只能写标识符
  • 就算引号省略了，键名也还是字符串，JavaScript会将属性名自动变成字符串
    • 1 -> ‘1’
    • 3.2 -> ‘3.2’
    • 1e2 -> ‘100’
    • 1e-2 -> ‘0.01’
    • .234 -> ‘0.234’
    • 0xFF -> ‘255’

删除对象的属性

let obj = {xxx:'yyy'}
//1. 删除属性的两种语法
delete obj.xxx
delete obj['xxx']
//2. 不包含属性xxx
'xxx' in obj === false // true
//3. 含有属性xxx，但是值为undefined
'xxx' in obj && obj.xxx === undefined  //false
//4. obj.xxx === undefined 不能断定‘xxx’是否为obj的属性
obj.xxx = undefined
obj.xxx === undefined //true

查看对象的属性

let obj = {k1:'v1',k2:'v2',k3:'v3'}
//1. 查看自身所有属性
Object.keys(obj)    //["k1", "k2", "k3"]
//2. 查看自身+共有属性
//2.1. 法1
console.dir(obj)
//2.2 法2
function getAllPropertyNames(obj){
    let temp = obj
    while(temp != null){
        console.log(Object.getOwnPropertyNames(temp))
        temp = temp.__proto__
    }
}
getAllPropertyNames(obj)
//3. 判断一个属性是自身的还是共有的
obj.hasOwnProperty('toString')  //false
// true: 自身的
// false: 共有的

//4， 查看单个属性的两种方法
obj['k1']
obj.k1

修改或增加对象的属性

let obj = {name: 'Peppa'}
//1. 修改属性
//1.1. 直接赋值
obj.name = 'Pig'
obj['name'] = 'Pig'
let key = name
obj[key] = 'Pig'
//1.2. 批量赋值
Object.assign(obj,{age:3,gender:'male'})

//2. 修改或增加共有属性
let obj1={},obj2={} //共有toString
obj1.__proto__.toString = 'xxx' //不推荐用__proto__
Object.prototype.toString = 'xxx'
obj2.toString //'xxx'

//3. 修改隐藏属性
//3.1. 直接修改__proto__ (不推荐)
let obj1 = {name:'Peppa'}
let obj2 = {name:'Pig'}
let common = {color:'pink'}
obj1.__proto__ = common
obj2.__proto__ = common
//3.2. 推荐使用Object.create
let obj1 = Object.create(common)
obj1.name = 'Peppa'
let obj2 = Object.create(common)
obj2.name = 'Pig'

‘name’ in obj和obj.hasOwnProperty(‘name’) 的区别

let common = {color:'pink'}
let obj = Object.create(common)
obj.name = 'Peppa'
console.log("'color' in obj: " + ('color' in obj))//true
console.log("'name' in obj: " + ('color' in obj))//true
console.log("obj.hasOwnProperty('color'): " + obj.hasOwnProperty('color'))//false
console.log("obj.hasOwnProperty('name'): " + obj.hasOwnProperty('name'))//true

‘name’ in obj 检查obj自身和共有属性

obj.hasOwnProperty(‘name’) 只检查obj自身属性
